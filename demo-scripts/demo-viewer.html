class="c">// =============================================================

class="c">// =============================================================

console.log(class="s">'╔══════════════════════════════════════════════╗');

class="c">// ── 1. Hiển thị toàn bộ mảng process.argv thô ────────────────
console.log(class="s">'📋 [process.argv] — Mảng đầy đủ (raw):');

  let label = class="s">'';
  if (index === 0) label = class="s">'← index[0]: đường dẫn node.exe';
  if (index === 1) label = class="s">'← index[1]: đường dẫn script(Entry Point)';
  if (index >= 2) label = class="t">`← index[${index}]: tham số người dùng`;
  console.log(class="t">`   [${index}] "${val}"  ${label}`);
});

class="c">// ── 2. Cắt bỏ 2 phần tử hệ thống, lấy args người dùng ───────
const userArgs = process.argv.slice(2);
class="c">//               └─ Phương pháp slice(2): loại bỏ node + script path
console.log(class="s">'\n✂️  [process.argv.slice(2)] — Tham số người dùng:');
console.log(class="s">'  ', userArgs);

class="c">// ── 3. Destructuring lấy sub-command ─────────────────────────
const [subCommand, ...restArgs] = userArgs;
console.log(class="s">'\n🎯 Sub-command(lệnh con):', subCommand || class="s">'(không có)');

class="c">// ── 4. Vòng lặp parse flag thủ công ──────────────────────────
const flags = {};
for (let i = 0; i < restArgs.length; i++) {
  const token = restArgs[i];
  if (token.startsWith(class="s">'--')) {
    const key = token.slice(2);           class="c">// Bỏ dấu "--"
    const nextToken = restArgs[i + 1];
    if (nextToken && !nextToken.startsWith(class="s">'--')) {
      class="c">// Tham số có giá trị: --quality 80
      flags[key] = nextToken;
      i++; class="c">// Bỏ qua token tiếp theo vì đã dùng làm giá trị
    } else {
      class="c">// Cờ boolean: --verbose(không có giá trị kèm theo)
      flags[key] = true;
    }
  }
}

console.log(class="s">'\n🚩 Flags đã parse(thủ công):');
console.table(flags);

class="c">// ── 5. Vấn đề của cách parse thủ công ────────────────────────
console.log(class="s">'⚠️  Vấn đề với parse thủ công:');
console.log(class="s">'   - "--quality 80": quality nhận giá trị kiểu STRING "80"');
console.log(class="s">'   - Chưa hỗ trợ alias: -i thay cho --input');

class="c">// =============================================================
class="c">// DEMO 4.1.2 (2) — Parse cấu hình đầu vào với commander
class="c">// Minh họa: Declarative API, requiredOption, alias, type coercion
class="c">//           tự sinh --help, sub-command action
class="c">// Chạy:  node 03-demo-commander-config.js compress -i ./images -o ./output -q 80
class="c">// Hoặc:  node 03-demo-commander-config.js --help
class="c">// =============================================================

import { Command } from class="s">'commander';

const program = new Command();

class="c">// ── 1. Metadata của CLI tool ──────────────────────────────────
program
  .name(class="s">'resize-img')
  .description(class="s">'Batch Image Resizer — CLI Tool(Node.js Core Demo)')
  .version(class="s">'1.0.0', class="s">'-v, --version', class="s">'Hiển thị phiên bản');

class="c">// ── 2. Định nghĩa sub-command "compress" ─────────────────────
program
  .command(class="s">'compress')
  .description(class="s">'Nén và resize hình ảnh hàng loạt trong thư mục')
  class="c">// requiredOption: bắt buộc, ném lỗi nếu người dùng bỏ qua
  .requiredOption(
    class="s">'-i, --input <directory>',
    class="s">'Đường dẫn thư mục chứa ảnh gốc(bắt buộc)'
  )
  class="c">// Option có giá trị mặc định
  .option(
    class="s">'-o, --output <directory>',
    class="s">'Đường dẫn thư mục lưu ảnh đã nén',
    class="s">'./output'
  )
  class="c">// Option với ép kiểu: chuỗi "80" → số 80 qua hàm parseInt
  .option(
    class="s">'-q, --quality <number>',
    class="s">'Chất lượng JPEG(1–100)',
    (val) => {
      const num = parseInt(val, 10);
      if (isNaN(num) || num < 1 || num > 100)
        throw new Error(class="s">'Quality phải là số nguyên từ 1 đến 100');
      return num;
    },
    80  class="c">// Giá trị mặc định
  )
  class="c">// Option với alias -W / --width
  .option(class="s">'-W, --width <pixels>', class="s">'Chiều rộng đích(pixels)', parseInt)
  .option(class="s">'-H, --height <pixels>', class="s">'Chiều cao đích(pixels)', parseInt)
  class="c">// Cờ boolean
  .option(class="s">'--verbose', class="s">'Hiển thị log chi tiết từng file', false)
  class="c">// Action: hàm được gọi khi sub-command khớp
  .action((options) => {
    console.log(class="s">'╔══════════════════════════════════════════════╗');
    console.log(class="s">'║     Demo: commander — Configuration Object   ║');
    console.log(class="s">'╚══════════════════════════════════════════════╝\n');

    console.log(class="s">'✅ Configuration Object đã parse(commander):');
    console.log({
      input:   options.input,
      output:  options.output,
      quality: options.quality,   class="c">// ← Đã là NUMBER, không phải string
      width:   options.width  || null,
      height:  options.height || null,
      verbose: options.verbose,
    });

    console.log(class="s">'\n📌 So sánh với parse thủ công:');
    console.log(class="s">'   quality(commander) :', typeof options.quality, class="s">'→', options.quality);
    console.log(class="s">'   → Ép kiểu tự động! Không cần parseInt thủ công\n');
    console.log(class="s">'   Alias hoạt động: -i ↔ --input | -o ↔ --output | -q ↔ --quality');
    console.log(class="s">'   Thử: node 03-demo-commander-config.js --help\n');
  });

class="c">// ── 3. Kích hoạt parser ───────────────────────────────────────
program.parse(process.argv);
