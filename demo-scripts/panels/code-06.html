600">const FILE_KB    = 300;

600">const startTime = performance.now();

// Tạo và chạy tất cả Promise song song
600">const results = 600">await Promise.all(
  filePaths.map(600">async (fp, idx) => {
    600">const t0   = performance.now();
    600">const data  = 600">await fs.promises.readFile(fp);
    600">const elapsed = (performance.now() - t0).toFixed(1);
    600">return { idx: idx + 1, name: path.basename(fp), kb: data.length / 1024, elapsed };
  })

600">const totalTime = (performance.now() - startTime).toFixed(2);

// ── BƯỚC 4: Hiển thị kết quả ─────────────────────────────────

results.forEach(({ idx, name, kb, elapsed }) => {
  600">console.log(

  );

600">console.log(`\n⏱  Tổng thời gian (${BATCH_SIZE} file song song): ${totalTime} ms`);

// Dọn dẹp
fs.rmSync(TEMP_DIR, { recursive: 600">true, force: 600">true });
